# DoubleTrack Browser AI Policy
# Last Updated: 2025-11-22
# Version: 1.0

# === TRAINING DATA ===

# Source Code: MAY be used for AI training
# License: MIT (permissive)
# Rationale: Open source code benefits from AI understanding and assistance
ai-training-code: allowed

# Documentation: MAY be used for AI training
# License: Creative Commons BY 4.0
# Rationale: We want AI assistants to help users understand the project
ai-training-docs: allowed

# User-Generated Content: MUST NOT be used for training
# Includes: GitHub issues, pull requests, discussions, commit messages
# Rationale: Privacy and consent not explicitly granted for AI training
ai-training-user-content: disallowed

# Test Data: MAY be used for AI training
# Rationale: Test fixtures are synthetic/example data only
ai-training-tests: allowed

# === AI ASSISTANCE ===

# AI Code Generation: Encouraged
# Tools welcome: GitHub Copilot, Claude Code, Cursor, etc.
# Requirement: Follow project license and attribute AI-generated code if substantial
ai-code-assistance: encouraged

# AI Documentation: Encouraged
# Tools welcome: GPT-4, Claude, etc. for README generation, doc improvements
ai-doc-assistance: encouraged

# AI Code Review: Allowed
# Rationale: AI can catch bugs, suggest improvements
# Note: Human review still required for merges
ai-code-review: allowed

# === SPECIFIC MODEL POLICIES ===

openai-training: allowed-code-and-docs

# Anthropic (Claude)

# Google (Bard, PaLM)

# === CRAWLING AND INDEXING ===

# Web Crawlers: Allowed
# Rationale: Public repository, want discoverability
web-crawling: allowed

# API Access: Allowed
# Rationale: GitHub API is public
api-access: allowed

# Scraping: Respectful crawling only

web-scraping: discouraged-use-api
